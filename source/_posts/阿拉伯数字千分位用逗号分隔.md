## 代码
```javascript
/*
*@param s 字符
*/
function setMoney(s){
    if(s.length < 3){
        return;
    }
    //判断是否有小数位 如果有,需要多加判断
    if(s.indexOf('.') > -1){
        var index = s.indexOf('.');
        var last = s.substr(index);
        s = s.substr(0,index).replace(/\B(?=(\d{3})+$)/g,',');
        s = s + last;
        return s;
    }else{
        //没有小数的就直接返回
        return s.replace(/\B(?=(\d{3})+$)/g,',')
    }
}
//测试
var str = '123123123123.56';
console.log(setMoney(str))
//123,123,123,123.56
```
