const transformParamToUrl = (param) => {
    const tempObj = []
    if (!param || typeof param !== 'object') {
        return
    }

    for (let i in param) {
        if (param.hasOwnProperty(i) && param[i]) {
            tempObj.push(i)
            tempObj.push('=')
            tempObj.push(param[i])
            tempObj.push('&')
        }
    }
    // remove the last field &
    tempObj.pop()
    return tempObj.join('')
}
